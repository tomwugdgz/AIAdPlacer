🐍 Python 调用
mcp_client.py 
import requests

# 通用 MCP 工具调用
def call_tool(name, arguments=None):
    url = "http://47.253.159.62:5002/api/v2/mcp/pdooh/tools/call"
    resp = requests.post(url, json={"name": name, "arguments": arguments or {}})
    return resp.json()

# 查询城市汇总
result = call_tool("pdooh_query_city_summary")
print(result)

# 查询智能屏
screens = call_tool("pdooh_query_screens", {
    "city": "广州", "district": "天河区", "limit": 10
})
print(screens)

# 人群洞察
insight = call_tool("pdooh_audience_insight", {
    "product_desc": "高端白酒", "target_city": "广州"
})
print(insight)
